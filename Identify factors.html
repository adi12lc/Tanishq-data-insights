def merge_filtered_with_sample(df):
    
    df_filtered = df[(df['action_type'] != 'view')& (df['action_type'] != 'add_to_cart')]
    
    df_view = df[df["action_type"] == 'view'].sample(n=3000, random_state=42)
    df_view2 = df[df["action_type"] == 'add_to_cart'].sample(n=3000, random_state=27)
    
    merged_df = pd.concat([df_filtered, df_view, df_view2])
    
    return merged_df

df = merge_filtered_with_sample(df)
