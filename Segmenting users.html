def plot_null_values(df):
    null_values = (round((df.isnull().sum() / len(df.index)) * 100, 2)).sort_values(ascending=False)
    
    if null_values.head(2)[1] == 0:
        print("No null values in this dataframe")
    else:
        plot_null = null_values.head(len([i for i in pd.isnull(df).sum() > 0 if i == True]))
        pd_null = pd.DataFrame(plot_null, columns=["% of null values"])
        sns.set(rc={'figure.figsize': (15, 8)})
        sns.set_style(style='white')
        sns.set_palette("Oranges")
        plt.title('Null values')
        plt.ylabel("Features")
        ax = sns.barplot(y=pd_null.index, x='% of null values', data=pd_null, palette="Oranges")
        ax.bar_label(ax.containers[0])

geolocator = Nominatim(user_agent="geoapiExercises")

def find_location_by_coordinates(lat, lon):
    try:
        location = geolocator.reverse([lat, lon], exactly_one=True)
        address = location.raw['address']
        city = address.get('city', '')
        if city == '':
            city = address.get('state', '')
        return city
    except GeocoderTimedOut:
        return find_location_by_coordinates(lat, lon)
    except:
        return ""

def parse_user_agent(user_agent_str):
    agent_info = httpagentparser.detect(user_agent_str)
    platform = agent_info.get('platform', {}).get('name', 'null')
    browser = agent_info.get('browser', {}).get('name', 'null')
    return platform, browser
